Architecture

Trois couches. Un chemin de données.

Code

Une seule ligne change.

client.ts · TypeScript 
// Drop-in from OpenAI SDK — only baseURL & tier change
import OpenAI from "openai";

const client = new OpenAI({
  baseURL: "https://api.infera.eu/v1",
  apiKey : process.env.INFERA_API_KEY,
  defaultHeaders: { "x-infera-tier": "trusted" },
});

const stream = await client.chat.completions.create({
  model: "mistral-large",
  stream: true,
  messages: [{ role: "user", content: "Bonjour, Europe." }],
});

for await (const part of stream) {
  process.stdout.write(part.choices[0].delta.content ?? "");
}
